'---------------------------------------------------------------------------------------------------------------------------------------
'Purpose - To Retrieve Data From Clip Board (Using IE)
'---------------------------------------------------------------------------------------------------------------------------------------

Set objIE = CreateObject("InternetExplorer.Application")
objIE.Navigate("about:blank")
strURL = objIE.document.parentwindow.clipboardData.GetData("text")
objIE.Quit
Wscript.Echo strURL
'---------------------------------------------------------------------------------------------------------------------------------------
