Public Function fn_GetName()
	If  UCase(g_HandleError) = "YES" Then
		On Error resume next 
	End If
			Set obj = fn_Object("SwfWindow","NaviLink")
			Set NaviLinkwin = SwfWindow(obj)
			sv_UserName_Temp =  NaviLinkwin.GetROProperty("text")
			If Len(sv_UserName_Temp) <> 0 Then
				sv_Split_UserName_Temp = Split(sv_UserName_Temp,": ")
				sv_Split_UserName = sv_Split_UserName_Temp(1)
				sv_Split_UserName_Temp1 = Split(sv_Split_UserName," ")
				sv_ArraySize = UBound(sv_Split_UserName_Temp1)
			
				sv_Temp_Name = sv_Split_UserName_Temp1(1)
				For i = 2 to sv_ArraySize
					sv_Temp_Name = sv_Temp_Name & " " & sv_Split_UserName_Temp1(i)
				Next
				sv_Temp_Name = sv_Temp_Name & ", " & sv_Split_UserName_Temp1(0)
				sv_Len  = Len(sv_Temp_Name)
				If sv_Len > 24  Then
					sv_FinalName = Left(sv_Temp_Name,24)
					fn_GetName = sv_FinalName
				Else
					fn_GetName = sv_Temp_Name	
				End If
			End If
End Function

'================================================================================================================
'
'				Creating the object
'
'================================================================================================================
Public function fn_Object(sv_Object,sv_Value)
   	If  UCase(g_HandleError) = "YES" Then
		On Error resume next 
	End If
	Set sv_objConnAdCn = CreateObject("ADODB.Connection")
	sv_strDataFile =  g_strMainPath & "\ObjectRepository\ObjectRepository.xls"
	sv_strConnection="Driver={Microsoft Excel Driver (*.xls)};DBQ=" & sv_strDataFile & ";ReadOnly=True;"
	sv_objConnAdCn.Open sv_strConnection
	sv_strSQL= "Select * from [" & sv_Object &"$] where ObjName = '" & sv_Value &"'" 
	Set sv_objRecAdRs = CreateObject("ADODB.Recordset")
	sv_objRecAdRs.Open sv_strSQL,sv_objConnAdCn
	sv_EOFFlag = 0
'If the configuration file record set is empty the exit run
'============================================
	If sv_objRecAdRs.EOF Then
		sv_EOFFlag = 1
		fn_UpdateErrorDetails "Input Error","Please check whether the following Object Name - " & sv_Value & " exist in the Sheet - " & sv_Object & ".","","fn_Object","","N","0","Y"
	End If
	
	If sv_EOFFlag = 0 Then
	' Moving to the First RecordSet
		sv_objRecAdRs.MoveFirst
'===============================================================================
		Do While not  sv_objRecAdRs.EOF
		
			Set obj = Description.Create()
			For sv_inc = 1 to 4
				sv_Paramtemp = sv_objRecAdRs("Prop" & sv_inc)
                sv_Valtemp = sv_objRecAdRs("Val" & sv_inc)
				If sv_Paramtemp <> "" Then
					obj(sv_Paramtemp).Value = sv_Valtemp
				End If
			Next
			
			Set fn_Object = obj
			Set obj = Nothing
			
			sv_objRecAdRs.MoveNext
		Loop
	End If
End Function

'Description : This function is used to Get Cell Data from the table
'Author :Benefits Team
''****************************************************************************************************************************************************************
Public Function SwfGetCell(obj,strSearchValue,strColName,sv_Value)
   	If  UCase(g_HandleError) = "YES" Then
		On Error resume next 
	End If
				   Dim intRowCount,blnFlag, strCellValue,i,strVal
				   blnFlag=0
				   If obj.Exist Then
					   Call WriteReport("Check for "& sv_Value & " Object",sv_Value & " Object Exist",sv_Value & " Object does Exist","P")
					   intRowCount =obj.RowCount
					   If strSearchValue=Empty Then
						   Call WriteReport("Check for "& sv_Value & " Object",sv_Value & "  Object Should not be Empty",sv_Value & " Object value is Empty","F")
                           Exit Function
					   Else
					   For i=0 to  intRowCount-1
						   sv_i = CInt(i)
							strCellValue=trim(obj.GetCellData(sv_i,strColName))
							If  LCase(Trim(strCellValue))=LCase(Trim(strSearchValue)) Then
										blnFlag=1
										SwfGetCell=strCellValue
										VerifyCellValue = i+1
										obj.SelectRow sv_i
										SwfGetCell = strCellValue
										Exit for
										End If
						Next
							If blnFlag=1 Then
							   Call WriteReport("Verify Value From Grid: "&strSearchValue & "Selected",sv_Value & " Selected",sv_Value & " Selected","P")
							Else
								   Call WriteReport("Verify Value From Grid: "&strSearchValue & " Selected",sv_Value & " Not Selected",sv_Value & " Not Selected","F")
									VerifyCellValue=0
						  End If
					 End If
					Else
							 Call WriteReport("Check for "& sv_Value & " Object",sv_Value & " Object Exist",sv_Value & " Object does not Exist","F")
								 VerifyCellValue=0
				End if	
End Function

'Function Name:fn_Scenarioresult()
'Description : This function is used to  for the reporting the execution details in the Email
'Author :Benefits Team
''****************************************************************************************************************************************************************
Public Function fn_Scenarioresult()
   If  UCase(g_HandleError) = "YES" Then
		On Error resume next 
	End If
	sv_strDataFile = g_strInputPath & "\INPUT.xls"

	Const ForReading = 1 
	Const ForWriting = 2 
	Const ForAppending = 8 
	Const TristateUseDefault = -2 
	Const TristateTrue = -1 
	Const TristateFalse = 0 
	
	Set sv_objConnAdCn = CreateObject("ADODB.Connection")
	sv_strConnection="Driver={Microsoft Excel Driver (*.xls)};DBQ=" & sv_strDataFile & ";ReadOnly=True;"
	sv_objConnAdCn.Open sv_strConnection

	sv_strSQL= "Select * from [USERINPUT$] where UCASE(TO_BE_EXECUTED) = 'YES'" 
	Set sv_objRecAdRs = CreateObject("ADODB.Recordset")
	sv_objRecAdRs.Open sv_strSQL,sv_objConnAdCn
	sv_EOFFlag = 0
	'If the configuration file record set is empty the exit run
	'============================================
	If sv_objRecAdRs.EOF Then
		sv_EOFFlag = 1
		On error goto 0 	
	End If
	
	If sv_EOFFlag = 0 Then
		' Moving to the First RecordSet
		sv_objRecAdRs.MoveFirst
		sv_temp_String = vbCrlf & vbTab & "Scenario Name" & vbTab & "Pass Count" & vbTab & "Fail Count" & vbTab & "Total Count"
	'===============================================================================
		Do While not  sv_objRecAdRs.EOF
			sv_Scenario = sv_objRecAdRs("SCENARIO")	

		   Set oFS = CreateObject("Scripting.FileSystemObject") 
		   sFilePathAndName = g_EmailDestinationPath  & sv_Scenario & ".html"
		   Set oFile = oFS.GetFile(sFilePathAndName) 
		   If oFS.FileExists(sFilePathAndName) = True Then
				Set oStream = oFile.OpenAsTextStream(ForReading, TristateUseDefault) 
				Do While Not oStream.AtEndOfStream 
					sRecord=oStream.ReadLine 
				   sv_WorkPackage_loci= instr(sRecord,"Work Package : ")
					If sv_WorkPackage_loci > 0 Then
						sv_temp_passcount = sv_WorkPackage_loci + 14
						sv_Passcount_loc1i = instr(sv_temp_passcount,sRecord," <DIV ")
						sv_templen = sv_Passcount_loc1i - sv_temp_passcount
						sv_PassCount_Val = mid(sRecord,sv_temp_passcount,sv_templen)
						sv_temp_String = sv_temp_String & vbCrlf & vbTab & sv_PassCount_Val
					End If
					sv_Passcount_loci= instr(sRecord,"Pass Count : ")
					If sv_Passcount_loci > 0 Then
						sv_temp_passcount = sv_Passcount_loci + 12
						sv_Passcount_loc1i = instr(sv_temp_passcount,sRecord," <DIV ")
						sv_templen = sv_Passcount_loc1i - sv_temp_passcount
						sv_PassCount_Val = mid(sRecord,sv_temp_passcount,sv_templen)
						sv_temp_String = sv_temp_String & vbTab & vbTab & sv_PassCount_Val
					End If
					sv_Failcount_loci= instr(sRecord,"Fail Count : ")
					If sv_Failcount_loci > 0 Then
						sv_temp_passcount = sv_Failcount_loci + 12
						sv_Passcount_loc1i = instr(sv_temp_passcount,sRecord," <DIV ")
						sv_templen = sv_Passcount_loc1i - sv_temp_passcount
						sv_PassCount_Val = mid(sRecord,sv_temp_passcount,sv_templen)
						sv_temp_String = sv_temp_String & vbTab & vbTab & sv_PassCount_Val
					End If
					sv_Totalcount_loci= instr(sRecord,"Total Count : ")
					If sv_Totalcount_loci > 0 Then
						sv_temp_passcount = sv_Totalcount_loci + 13
						sv_Passcount_loc1i = instr(sv_temp_passcount,sRecord," <DIV ")
						sv_templen = sv_Passcount_loc1i - sv_temp_passcount
						sv_PassCount_Val = mid(sRecord,sv_temp_passcount,sv_templen)
						sv_temp_String = sv_temp_String & vbTab & vbTab & sv_PassCount_Val' & vbCrlf
						Exit Do
					End If
				 Loop 
			End If
			 oStream.Close 
			 sv_objRecAdRs.MoveNext
		Loop
	 End If
	 fn_Scenarioresult = sv_temp_String
End Function
